165  typedef enum PThreadStatus {
166  THREADLIB_BUILDED,
167  THREADLIB_INITIALIZED,
168  THREADLIB_STARTED, // and not joined
169  THREADLIB_DETACHED,
170 
171  // When a POSIX thread is created, the operative system
172  // frees the relative resources when the thread is joined
173  // or is detached.
174  // In case the thread will not be joined and will not be
175  // detached, the resources are not freed and
176  // it will cause a memory leak. The next state is added
177  // to be sure that there will not be memory leak. In fact, when
178  // the thread is finishing and is not joined or detached (state
179  // THREADLIB_STARTED or THREADLIB_DETACHED),
180  // it will be detached automatically.
181  THREADLIB_STARTED_AND_JOINED,
182 
183  // When the thread is finished, the state returns
184  // to INITIALIZED. The next state is added to discriminate
185  // the INITIALIZED state after the 'init' method and
186  // the INITIALIZED state when the thread is finished.
187  THREADLIB_INITIALIZEDAGAINAFTERRUNNING
188  } PThreadStatus_t, *PThreadStatus_p;
