67  typedef enum PThreadStatus {
68  THREADLIB_BUILDED,
69  THREADLIB_INITIALIZED,
70  THREADLIB_STARTED, // and not joined
71  THREADLIB_DETACHED,
72 
73  // When a POSIX thread is created, the operative system
74  // frees the relative resources when the thread is joined
75  // or is detached.
76  // In case the thread will not be joined and will not be
77  // detached, the resources are not freed and
78  // it will cause a memory leak. The next state is added
79  // to be sure that there will not be memory leak. In fact, when
80  // the thread is finishing and is not joined or detached (state
81  // THREADLIB_STARTED or THREADLIB_DETACHED),
82  // it will be detached automatically.
83  THREADLIB_STARTED_AND_JOINED,
84 
85  // When the thread is finished, the state returns
86  // to INITIALIZED. The next state is added to discriminate
87  // the INITIALIZED state after the 'init' method and
88  // the INITIALIZED state when the thread is finished.
89  THREADLIB_INITIALIZEDAGAINAFTERRUNNING
90  } PThreadStatus_t, *PThreadStatus_p;
